  <button onclick="checkCustomer()">조회하기</button>


  <div id="result"></div>


  <script>

    async function checkCustomer() {

      const name = document.getElementById("name").value;

      const phone = document.getElementById("phone").value;


      const resultDiv = document.getElementById("result");

      resultDiv.innerHTML = "조회중...";


      try {

        const response = await fetch("/.netlify/functions/getCustomer", {

          method: "POST",

          headers: {

            "Content-Type": "application/json"

          },

          body: JSON.stringify({ name, phone })

        });


        const data = await response.json();


        if (response.status !== 200) {

          resultDiv.innerHTML = "❌ " + data.error;

          return;

        }


        resultDiv.innerHTML = `

          <p><strong>선불권:</strong> ${data.prepaid_name}</p>

          <p><strong>총 회차:</strong> ${data.total}</p>

          <p><strong>사용 횟수:</strong> ${data.used}</p>

          <p><strong>잔여 횟수:</strong> ${data.remain}</p>

          <p><strong>만료일:</strong> ${data.expire}</p>

        `;

      } catch (error) {

        resultDiv.innerHTML = "서버 오류 발생";

      }

    }

  </script>
